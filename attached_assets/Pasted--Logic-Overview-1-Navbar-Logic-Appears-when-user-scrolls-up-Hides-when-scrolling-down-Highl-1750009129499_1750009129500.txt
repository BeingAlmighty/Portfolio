// Navbar.jsx
import { useEffect, useState } from 'react';

export default function Navbar() {
  const [show, setShow] = useState(true);
  let lastScroll = 0;

  useEffect(() => {
    const handleScroll = () => {
      const current = window.scrollY;
      setShow(current < lastScroll || current < 100);
      lastScroll = current;
    };
    window.addEventListener('scroll', handleScroll);
    return () => window.removeEventListener('scroll', handleScroll);
  }, []);

  return (
    <nav
      className={`fixed top-5 left-1/2 transform -translate-x-1/2 px-8 py-3 z-50 rounded-full backdrop-blur-md bg-white/10 border border-white/20 transition-all duration-300 ${
        show ? 'opacity-100' : 'opacity-0 -translate-y-5'
      }`}
    >
      <ul className="flex gap-6 text-white text-sm font-semibold">
        <li><a href="#home">Home</a></li>
        <li><a href="#services">Services</a></li>
        <li><a href="#projects">Projects</a></li>
        <li><a href="#about">About Us</a></li>
        <li><a href="#contact">Contact</a></li>
      </ul>
    </nav>
  );
}

// Hero.jsx
import { motion, useScroll, useTransform } from 'framer-motion';

export default function Hero() {
  const { scrollY } = useScroll();
  const zoom = useTransform(scrollY, [0, 500], ['150%', '100%']);
  const fade = useTransform(scrollY, [0, 500], [1, 0]);

  return (
    <motion.section
      id="home"
      className="h-screen w-full bg-cover bg-center flex items-center justify-center"
      style={{
        backgroundImage: "url('https://source.unsplash.com/1600x900/?tech')",
        backgroundSize: zoom,
      }}
    >
      <motion.div style={{ opacity: fade }} className="text-center text-white">
        <h1 className="text-5xl font-bold">Welcome to Our Agency</h1>
        <p className="mt-4 text-xl">We build stunning digital experiences.</p>
      </motion.div>
    </motion.section>
  );
}

// Services.jsx
import { motion } from 'framer-motion';

export default function Services() {
  const services = [
    { title: 'Web Development', desc: 'Modern, fast, scalable websites' },
    { title: 'UI/UX Design', desc: 'Beautiful, intuitive interfaces' },
    { title: 'SEO Optimization', desc: 'Rank higher in search results' },
  ];

  return (
    <section id="services" className="py-20 bg-black text-white">
      <h2 className="text-4xl font-bold text-center mb-12">Our Services</h2>
      <div className="grid md:grid-cols-3 gap-8 px-8 max-w-6xl mx-auto">
        {services.map((service, i) => (
          <motion.div
            key={i}
            whileInView={{ opacity: 1, y: 0 }}
            initial={{ opacity: 0, y: 50 }}
            transition={{ duration: 0.6, delay: i * 0.2 }}
            className="relative w-full h-72 perspective"
          >
            <div className="w-full h-full transition-transform duration-500 transform-style-3d hover:rotate-y-180 bg-white text-black rounded-xl shadow-lg">
              <div className="p-6">
                <h3 className="text-xl font-semibold">{service.title}</h3>
                <p className="mt-2">{service.desc}</p>
              </div>
              <div className="absolute inset-0 p-6 rotate-y-180 backface-hidden bg-black text-white flex items-center justify-center rounded-xl">
                <p className="text-sm">More details coming soon...</p>
              </div>
            </div>
          </motion.div>
        ))}
      </div>
    </section>
  );
}
