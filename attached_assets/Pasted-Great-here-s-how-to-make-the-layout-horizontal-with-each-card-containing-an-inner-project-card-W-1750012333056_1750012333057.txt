      title: 'Food Ordering App',

    {
      color: 'green',
      title: 'Portfolio Website',
      desc: 'Personal brand using Next.js.',
      img: 'https://source.unsplash.com/300x200/?developer,portfolio',
    },
    {
      color: 'yellow',
      title: 'Task Manager',
      desc: 'Task assignment + leaderboard.',
      img: 'https://source.unsplash.com/300x200/?tasks,planner',
    },
  ];

  return (
    <div className="cards">
      {projects.map((proj, index) => (
        <div className={`card ${proj.color}`} key={index}>
          <div className="project">
            <img src={proj.img} alt={proj.title} />
