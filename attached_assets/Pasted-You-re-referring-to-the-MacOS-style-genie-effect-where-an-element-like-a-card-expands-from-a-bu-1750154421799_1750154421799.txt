import { useState } from "react";
import { motion, AnimatePresence } from "framer-motion";

export default function GenieEffect() {
  const [isOpen, setIsOpen] = useState(false);

  return (
    <div className="flex justify-center items-center h-screen bg-neutral-900">
      <AnimatePresence>
        {!isOpen && (
          <motion.button
            initial={{ scale: 0.9 }}
            animate={{ scale: 1 }}
            whileHover={{ scale: 1.05 }}
            whileTap={{ scale: 0.95 }}
            className="bg-white text-black font-semibold px-6 py-3 rounded-lg shadow-xl"
            onClick={() => setIsOpen(true)}
          >
            Show Card
          </motion.button>
        )}

        {isOpen && (
          <motion.div
            key="card"
            initial={{
              opacity: 0,
              scale: 0.5,
              y: -100,
              originX: 0.5,
              originY: 0.5,
            }}
            animate={{
              opacity: 1,
              scale: 1,
              y: 0,
              transition: { type: "spring", stiffness: 100, damping: 15 },
            }}
            exit={{
              opacity: 0,
              scale: 0.3,
              y: -100,
              transition: { duration: 0.4, ease: "easeInOut" },
            }}
            className="relative bg-white text-black w-[300px] h-[200px] rounded-xl shadow-2xl flex flex-col items-center justify-center"
          >
            <h2 className="text-xl font-bold mb-4">Card Title</h2>
            <p className="text-gray-700 mb-6 px-4 text-center">
              This is the content of the card. Smooth animation, just like macOS.
            </p>
            <button
              className="bg-black text-white px-4 py-2 rounded-md"
              onClick={() => setIsOpen(false)}
            >
              Close
            </button>
          </motion.div>
        )}
      </AnimatePresence>
    </div>
  );
}

‚ú® Animation Highlights
initial ‚Üí sets starting transform when showing

animate ‚Üí springy pop-out for genie appearance

exit ‚Üí scale-down + fade for sucking back effect

AnimatePresence ‚Üí handles mount/unmount animation
